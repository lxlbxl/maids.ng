function toggleMobileMenu() { const menu = document.getElementById('mobile-menu'); if (menu.classList.contains('translate-y-full')) { menu.classList.remove('translate-y-full'); menu.classList.add('translate-y-0'); document.body.style.overflow = 'hidden'; } else { menu.classList.remove('translate-y-0'); menu.classList.add('translate-y-full'); document.body.style.overflow = 'auto'; } } const apiService = new ApiService('http://localhost:8000/api'); let currentPage = 1; // Initialize mobile menu button if exists document.addEventListener('DOMContentLoaded', () => { const btn = document.getElementById('mobile-menu-btn'); if (btn) btn.addEventListener('click', toggleMobileMenu); const close = document.getElementById('close-menu'); if (close) close.addEventListener('click', toggleMobileMenu); }); // Initialize state document.addEventListener('DOMContentLoaded', async () => { const token = localStorage.getItem('access_token'); // Basic auth check if (!token) { window.location.href = 'login.html'; return; } apiService.token = token; // Initial load await Promise.all([ loadDashboard(), loadMaids(1), loadProfile() ]); // Setup Tab Switching logic properly setupTabs(); }); function setupTabs() { document.querySelectorAll('[id^="tab-"]').forEach(tab => { tab.addEventListener('click', () => { const targetId = tab.id.replace('tab-', ''); switchTab(targetId); }); }); // Mobile menu toggle document.getElementById('mobile-menu-btn').addEventListener('click', function() { const menu = document.getElementById('mobile-menu'); menu.classList.toggle('hidden'); }); } async function loadDashboard() { try { const response = await apiService.get('/agency/dashboard'); if (response.success) { const stats = response.stats || {}; document.getElementById('stat-total-maids').textContent = stats.total_maids || 0; document.getElementById('stat-active-placements').textContent = stats.active_placements || 0; document.getElementById('stat-earnings').textContent = '₦' + (stats.total_earnings || 0).toLocaleString(); document.getElementById('agency-name').textContent = stats.agency_name || 'Agency'; renderRecentMaids(response.recent_maids); } } catch (error) { console.error('Error loading dashboard:', error); if (error.message === 'Unauthorized' || error.status === 401) logout(); } } async function loadMaids(page = 1) { try { const response = await apiService.get(`/agency/maids?page=${page}&limit=10`); if (response.success) { renderMaidsTable(response.data); updatePagination(response.meta); currentPage = page; } } catch (error) { console.error('Error loading maids:', error); } } async function loadProfile() { try { const response = await apiService.get('/agency/profile'); if (response.success && response.profile) { const p = response.profile; document.getElementById('profile-name').value = p.agency_name || ''; document.getElementById('profile-email').value = p.email || ''; document.getElementById('profile-phone').value = p.phone || ''; document.getElementById('profile-address').value = p.address || ''; } } catch (error) { console.error('Error loading profile:', error); } } // --- Render Functions --- function renderRecentMaids(maids) { const container = document.getElementById('recent-maids-list'); if (!maids || maids.length === 0) { container.innerHTML = '
No workforce records yet
'; return; } container.innerHTML = maids.map(maid => `

${maid.full_name}

${maid.work_type} • ${maid.location}

`).join(''); } function renderMaidsTable(maids) { const tbody = document.getElementById('maids-table-body'); if (!maids || maids.length === 0) { tbody.innerHTML = '

No Records Found

Register personnel to populate directory

'; return; } tbody.innerHTML = maids.map(maid => `
${maid.full_name}
${maid.age || '?'} yrs • ${maid.gender || 'Personnel'}
${maid.availability_status || 'Unknown'}
₦${(maid.salary_min || 0).toLocaleString()}
Gross Monthly
${maid.work_type}
${maid.location}
`).join(''); } function updatePagination(meta) { if (!meta) return; document.getElementById('page-info').textContent = `Page ${meta.current_page} of ${meta.last_page}`; document.getElementById('prev-page-btn').disabled = meta.current_page <= 1; document.getElementById('next-page-btn').disabled = meta.current_page >= meta.last_page; } // --- Actions --- function changePage(delta) { loadMaids(currentPage + delta); } function switchTab(tabId) { ['overview', 'maids', 'profile'].forEach(id => { const btn = document.getElementById(`tab-${id}`); const view = document.getElementById(`view-${id}`); const indicator = btn.querySelector('.h-1'); if (id === tabId) { btn.classList.add('tab-active'); btn.classList.remove('tab-inactive'); indicator.classList.remove('bg-transparent'); indicator.classList.add('bg-teal-600'); view.classList.remove('hidden'); } else { btn.classList.add('tab-inactive'); btn.classList.remove('tab-active'); indicator.classList.remove('bg-teal-600'); indicator.classList.add('bg-transparent'); view.classList.add('hidden'); } }); } function logout() { apiService.clearTokens(); localStorage.removeItem('userPhone'); localStorage.removeItem('userData'); window.location.href = 'login.html'; } // --- Modal Actions --- function openMaidModal(maid = null) { const modal = document.getElementById('maid-modal'); const title = document.getElementById('modal-title'); const form = document.getElementById('maid-form'); form.reset(); // Clear previous data if (maid) { title.textContent = 'Edit Maid'; document.getElementById('maid-id').value = maid.id; document.getElementById('maid-name').value = maid.full_name; document.getElementById('maid-age').value = maid.age; document.getElementById('maid-gender').value = maid.gender; document.getElementById('maid-phone').value = maid.phone || ''; document.getElementById('maid-work-type').value = maid.work_type; document.getElementById('maid-salary').value = maid.salary_min; document.getElementById('maid-location').value = maid.location; document.getElementById('maid-bio').value = maid.bio || ''; } else { title.textContent = 'Add New Maid'; document.getElementById('maid-id').value = ''; } modal.classList.remove('hidden'); } function closeMaidModal() { document.getElementById('maid-modal').classList.add('hidden'); } function editMaid(maid) { openMaidModal(maid); } async function deleteMaid(id) { if (!confirm('Are you sure you want to remove this maid?')) return; try { await apiService.delete(`/agency/maids/${id}`); loadMaids(currentPage); // Reload loadDashboard(); // Update stats } catch (error) { console.error('Delete error:', error); alert('Failed to delete maid'); } } // --- Forms --- document.getElementById('maid-form').addEventListener('submit', async (e) => { e.preventDefault(); const btn = document.getElementById('maid-submit-btn'); const id = document.getElementById('maid-id').value; const data = { full_name: document.getElementById('maid-name').value, age: document.getElementById('maid-age').value, gender: document.getElementById('maid-gender').value, phone: document.getElementById('maid-phone').value, work_type: document.getElementById('maid-work-type').value, salary_min: document.getElementById('maid-salary').value, location: document.getElementById('maid-location').value, bio: document.getElementById('maid-bio').value }; btn.disabled = true; btn.textContent = 'Saving...'; try { if (id) { await apiService.put(`/agency/maids/${id}`, data); } else { await apiService.post('/agency/maids', data); } closeMaidModal(); loadMaids(currentPage); loadDashboard(); } catch (error) { console.error('Save error:', error); alert('Failed to save maid: ' + (error.message || 'Unknown error')); } finally { btn.disabled = false; btn.textContent = 'Save Maid'; } }); document.getElementById('profile-form').addEventListener('submit', async (e) => { e.preventDefault(); const btn = document.getElementById('profile-submit-btn'); const data = { agency_name: document.getElementById('profile-name').value, email: document.getElementById('profile-email').value, phone: document.getElementById('profile-phone').value, address: document.getElementById('profile-address').value, pin: document.getElementById('profile-pin').value || undefined }; btn.disabled = true; btn.textContent = 'Saving...'; try { await apiService.put('/agency/profile', data); alert('Profile updated successfully'); document.getElementById('profile-pin').value = ''; // Clear pin } catch (error) { console.error('Profile update error:', error); alert('Failed to update profile'); } finally { btn.disabled = false; btn.textContent = 'Save Changes'; } });
